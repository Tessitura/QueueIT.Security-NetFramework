[KnownUser]
public ActionResult Target(KnownUserModel knownUserModel)
{
    PersistModel model = new PersistModel(
        knownUserModel.KnownUser.QueueId,
        knownUserModel.KnownUser.PlaceInQueue,
        knownUserModel.KnownUser.TimeStamp);

    model.Persist();

    return View();
}
